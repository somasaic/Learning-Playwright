42 - Switch Basics

42_Switch.js Switch

A basic switch statement that maps day numbers (0-6) to day names. Each case includes a break statement to prevent fall-through, and a default case handles invalid values.

// Switch
// 0 - Sunday, 1 - Monday, 2 - Tue.....
let day = 2;
switch (day) {
    case 0:
        console.log("Sunday — Rest Day");
        let a = 10;
        let b = 30;
        console.log(a + b);
        break;
    case 1:
        console.log("Monday — Sprint Planning");
        break;
    case 2:
        console.log("Tuesday — Development");
        break;
    case 3:
        console.log("Wednesday — Code Review");
        break;
    case 4:
        console.log("Thursday — Testing");
        break;
    case 5:
        console.log("Friday — Deployment & Retro");
        break;
    case 6:
        console.log("Saturday — Rest Day");
        break;
    default:
        console.log("Invalid day value");
}

45 - Real Example: API Status Codes

45_Switch_REAL_EXAMPLE.js API Validation

A practical example using switch to handle HTTP response codes during API validation - a common pattern in test automation.

// You are working API Validation
// response Code - 200, 404, 401, 403.....404

let responseCode = 404;

switch (responseCode) {

    case 200:
        console.log("200 Ok");
        break;
    case 404:
        console.log("404 Not found!");
        break;
    default:
        console.log("Not status code match");

}

46 - Group Case (Multiple Cases, Same Action)

46_Switch_GroupCase.js Grouped Cases

Group multiple cases together to execute the same block. Here, Chrome, Edge, Brave, and Opera all share the Chromium engine, so they share the same case output.

let browser = "Edge";

switch (browser) {
    case "Chrome":
    case "Edge":
    case "Brave":
    case "Opera":
        console.log("Chromium Project!");
        break;
    case "Firefox":
        console.log("Mozilla Project!");
        break;
    case "Safari":
        console.log("Apple browser — uses JavaScriptCore engine");
        break;
    default:
        console.log("Unknown browser — manual testing needed");

}

47 - IQ: Fall-through Bug

47_IQ_BUG.js Interview Question

A common interview question: What happens when break is missing? With fruit = "banana", it matches case "banana" and falls through all remaining cases including default.

let fruit = "banana";
switch (fruit) {
    case "apple":
        console.log("Apple selected");
    case "banana":
        console.log("Banana selected");
    case "cherry":
        console.log("Cherry selected");
    case "date":
        console.log("Date selected");
    default:
        console.log("Default reached");
}

48 - IQ: Switch with Range (switch true)

48_IQ.js Interview Question

Using switch(true) to evaluate range conditions. Each case is a boolean expression, and the first one that evaluates to true is matched.

let testScore = 85;

switch (true) {
    case (testScore >= 95):
        console.log("Outstanding — Top performer");
        break;
    case (testScore >= 85):
        console.log("Excellent — Above expectations");
        break;
    case (testScore >= 70):
        console.log("Good — Meets expectations");
        break;
    case (testScore >= 50):
        console.log("Needs Improvement");
        break;
    default:
        console.log("Unsatisfactory — Requires training");
}

50 - IQ: Strict Comparison (===)

50_Switch_IQ.js Interview Question

Switch uses strict comparison (===), so "5" (string) does not match 5 (number). Type matters!

let value = "5";

switch (value) {
    case 5:
        console.log("Number 5 matched");
        break;
    case "5":
        console.log("String '5' matched");
        break;
}
// Output: "String '5' matched"
// switch uses ===, so "5" !== 5 (different types)

51 - IQ: Type Coercion with 0 and false

51_Switch_IQ.js Interview Question

Another strict comparison example: 0 === false is false because they are different types (number vs boolean).

let status = 0;
switch (status) {
    case false:
        console.log("false matched");
        break;
    case 0:
        console.log("0 matched");
        break;
}

// Output: "0 matched" (0 === 0, NOT 0 === false)
