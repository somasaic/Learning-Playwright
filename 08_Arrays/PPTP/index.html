65 - Adding & Removing Elements

65_Arrays_Adding_Remove.js push/pop/shift/unshift/splice

Methods for adding and removing elements: push/pop (end), unshift/shift (beginning), and splice (any position).

let arr = [1, 2, 3];
console.log(arr);
// Add to END
arr.push(4);
console.log(arr);

// Remove from END
arr.pop();
console.log(arr);

arr.push(5, 6);
console.log(arr);


// Add to BEGINNING
arr.unshift(0);
console.log(arr);

// Remove from BEGINNING
arr.shift();
console.log(arr);

// [ 1, 2, 3, 5, 6 ]

// splice(start, deleteCount, ...itemsToAdd)
arr.splice(2, 1); // removes 1 item at index 2
console.log(arr);

arr.splice(2, 0, 99);
console.log(arr);

arr.splice(1, 2, 10, 20);
console.log(arr);

67 - Searching Arrays

67_Array_Searching.js Search Methods

Search methods: indexOf, lastIndexOf, includes, find, findIndex, findLast, and findLastIndex.

//Searching

let results = ["pass", "fail", "pass", "error", "fail"];

//// indexOf — returns first index, or -1 if not found

results.indexOf("fail"); //1
results.indexOf("skip");        // -1

// lastIndexOf — searches from the end
results.lastIndexOf("fail");    // 4

// includes — returns boolean
results.includes("error"); // true
results.includes("skip");       // false



// find — returns first matching element

let nums = [10, 25, 30, 45];
nums.find(x => x > 20);  // 25,30,45 -> A -25

// findIndex
nums.findIndex(n => n > 20);

nums.findLast(n => n > 20); // 45]
nums.findLastIndex(n => n > 20);// 3

68 - Iterating Arrays

68_Arrays_Iterating.js Iteration

Multiple ways to iterate: classic for, for...of (values), forEach (callback), entries() (index + value), and for...in (indexes).

// Iterate - Go from one to another. //

let tests = ["login", "checkout", "search"];

for (let i = 0; i < tests.length; i++) {
    console.log(tests[i]);
}

console.log("----");

// for...of (cleanest for values)
for (let test of tests) {
    console.log(test) // value
}
console.log("----");

// forEach (no return value)
tests.forEach((test, index) => {

    console.log(`${index}: ${test}`);
});

console.log("----");
// entries() — index + value

for (let [i, test] of tests.entries()) {
    console.log(i, test);
}

console.log("----");

let students = ["methis", "senthil", "ajay", "rahul"];


for (let student in students) {
    console.log(student, " -> ", students[student]); // index = in
}

69 - Transforming Arrays

69_Arrays_Transforming_Arrays.js map/filter/reduce/flat

Transformation methods: map (transform each element), filter (keep matching elements), reduce (accumulate to single value), and flat (flatten nested arrays).

let scores = [45, 82, 91, 60, 73];

// map — transforms every element, returns NEW array

let grades = scores.map(s => s > 70 ? "Pass" : "Fail");
console.log(grades);


// filter — keeps elements that pass a test

let passing = scores.filter(s => s >= 70);
console.log(passing);

// reduce — accumulates to a single value

let total = scores.reduce((sum, s) => sum + s, 0); //351
console.log(total);

// flat — flattens nested arrays
let nested = [[1, 2], [3, 4], [5]];
console.log(nested.flat());

70 - Sorting Arrays

70_Array_Sorting.js Sorting

Default sort() compares as strings (which can give wrong results for numbers). Use a comparator function for numeric sorting.

//Sorting

let fruits = ["banana", "apple", "cherry"];
fruits.sort();
console.log(fruits);
//  alphabetical by default

let nums = [10, 1, 21, 2];
nums.sort(); // [1, 10, 2, 21] — WRONG (compares as strings!)
console.log(nums);
nums.sort((a, b) => a - b); // Ascending
console.log(nums);
nums.sort((a, b) => b - a);
console.log(nums); // [21, 10, 2, 1] — descending

72 - Checking Arrays

72_Arrays_Checking.js isArray/every/some

Check if a value is an array with Array.isArray(). Use every() to check if all elements pass a test, and some() to check if at least one passes.

// Checking Arrays

// Check if something IS an array


let result = Array.isArray([1, 2, 3]);
let result1 = Array.isArray("a");
console.log(result);
console.log(result1);


// every & some
[80, 90, 85].every(s => s >= 70);   // true
[80, 60, 85].every(s => s >= 70);   // false

// some — AT LEAST ONE must pass
[80, 60, 85].some(s => s < 70); //  true
[80, 90, 85].some(s => s < 70); // false

73 - Copying Arrays (Shallow & Deep)

73_Arrays_Copying_Shallow_Deep.js Copy Methods

Shallow copy methods: spread, slice(), Array.from(), concat(). Direct assignment creates a reference (not a copy), so changes affect both variables.

let original = [1, 2, 3];

let copy1 = [...original]; // spread
// console.log(copy1);
let copy2 = original.slice();
// console.log(copy2);
let copy3 = Array.from(original);
// console.log(copy3);
let copy4 = original.concat();
// console.log(copy4);

copy1.push(99);
console.log(original);
console.log(copy1);

// Deep copy (JSON)
let c = original; // Deep copy
original.push(99);
console.log(original);
console.log(copy1);
